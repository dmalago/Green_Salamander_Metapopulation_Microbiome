### converts representative sequence file from Zymo into a qiime-accessible file :D

qiime tools import \
  --input-path sv.seqs.fna \
  --output-path rep-seqs.qza \
  --type 'FeatureData[Sequence]'

# will create rooted + unrooted phylogenetic trees of representative sequences
qiime phylogeny align-to-tree-mafft-fasttree \
--i-sequences rep-seqs.qza \
--o-alignment aligned-rep-seqs.qza \
--o-masked-alignment masked-aligned-rep-seqs.qza \
--o-tree unrooted-tree.qza \
--o-rooted-tree rooted-tree.qza

#write the trees out into newick format
qiime tools export --input-path rooted-tree.qza --output-path rooted_tree_out
qiime tools export --input-path unrooted-tree.qza --output-path unrooted_tree_out
